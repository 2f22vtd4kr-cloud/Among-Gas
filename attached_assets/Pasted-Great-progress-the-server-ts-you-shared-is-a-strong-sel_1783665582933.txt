  ws.on('message', (raw: Buffer) => {

    if (!ws.isVerified) {
      const user = verifyTelegramAuth(raw.toString());
      if (user) {
        ws.isVerified = true;
        ws.tgUserId = user.id;
        ws.username = user.username || `User${user.id}`;
        ws.send(Buffer.from([0x01, 0])); // Success placeholder
