import { WebSocketServer, WebSocket } from 'ws';
import { createHmac } from 'crypto';
import { URLSearchParams } from 'url';

// ==========================================
// 1. DATA MODELS & INTERFACES (SPEC REFERENCES)
// ==========================================

export enum GameState {
  LOBBY = 0x00,
  SPAWN = 0x01,
  ROAMING = 0x02,
  DISCUSSION = 0x03,
  VOTING = 0x04,
  EJECTION = 0x05,
  GAMEOVER = 0x06
}

export interface Player {
  id: number;               // Server slot index (0 - 14)
  tgUserId: number;         // Authenticated Telegram ID
  username: string;         // Telegram handle
  isHost: boolean;
  role: 'crewmate' | 'impostor';
  alive: boolean;
  isGhost: boolean;
  x: number;                // Authoritative X coordinate
  y: number;                // Authoritative Y coordinate
  lastBroadcastX: number;   // Cache to compute threshold deltas
  lastBroadcastY: number;
}

export class Lobby {
  public code: string;
  public hostSlotId: number = 0;
  public state: GameState = GameState.LOBBY;
  public players: Map<number, Player> = new Map(); // Keyed by slot ID (0-14)
  
  constructor(code: string) {
    this.code = code;
  }

  public getFirstAvailableSlot(): number {
    for (let i = 0; i < 15; i++) {
      if (!this.players.has(i)) return i;
    }
    return -1;
  }
}

interface AuthenticatedSocket extends WebSocket {
  isVerified: boolean;
  tgUserId?: number;
  username?: string;
  lobbyCode?: string;
  playerSlotId?: number;
}

// ==========================================
// 2. SECURE TELEGRAM AUTH ENGINE
// ==========================================

interface TelegramUser {
  id: number;
  first_name: string;
  username?: string;
}

function verifyTelegramAuth(initData: string, botToken: string): TelegramUser | null {
  try {
    const urlParams = new URLSearchParams(initData);
    const hash = urlParams.get('hash');
    if (!hash) return null;

    urlParams.delete('hash');

    // Sort parameters alphabetically to construct validation signature match
    const dataCheckArr: string[] = [];
    urlParams.forEach((val, key) => dataCheckArr.push(`${key}=${val}`));
    dataCheckArr.sort();
    const dataCheckString = dataCheckArr.join('\n');

    // Create Telegram HMAC SHA256 Key Chain
    const secretKey = createHmac('sha256', 'WebAppData').update(botToken).digest();
    const calculatedHash = createHmac('sha256', secretKey).update(dataCheckString).digest('hex');

    if (calculatedHash !== hash) {
      return null; // Cryptographic validation failed
    }

    const userString = urlParams.get('user');
    if (!userString) return null;

    return JSON.parse(userString) as TelegramUser;
  } catch (error) {
    console.error('Handshake verification exception:', error);
    return null;
  }
}

// ==========================================
// 3. LOBBY & MATCHMAKING LIFECYCLE MANAGER
// ==========================================

export class LobbyManager {
  public lobbies: Map<string, Lobby> = new Map();
  public userToLobbyMap: Map<number, string> = new Map();
  private alphabet = 'ABCDEFGHJKLMNPQRSTUVWXYZ23456789';

  public generateUniqueCode(): string {
    let code = '';
    do {
      code = '';
      for (let i = 0; i < 6; i++) {
        code += this.alphabet.charAt(Math.floor(Math.random() * this.alphabet.length));
      }
    } while (this.lobbies.has(code));
    return code;
  }

  public createLobby(tgUserId: number, username: string): Lobby {
    const code = this.generateUniqueCode();
    const lobby = new Lobby(code);
    
    const player: Player = {
      id: 0, // Host is assigned first index slot
      tgUserId,
      username,
      isHost: true,
      role: 'crewmate',
      alive: true,
      isGhost: false,
      x: 0, y: 0, lastBroadcastX: 0, lastBroadcastY: 0
    };

    lobby.players.set(0, player);
    lobby.hostSlotId = 0;
    
    this.lobbies.set(code, lobby);
    this.userToLobbyMap.set(tgUserId, code);
    
    console.log(`Lobby ${code} created by TG User: ${username}`);
    return lobby;
  }

  public joinLobby(code: string, tgUserId: number, username: string): { lobby: Lobby; slotId: number } | string {
    const formattedCode = code.toUpperCase().trim();
    const lobby = this.lobbies.get(formattedCode);

    if (!lobby) return "ERR_NOT_FOUND";
    if (lobby.state !== GameState.LOBBY) return "ERR_IN_PROGRESS";
    if (lobby.players.size >= 15) return "ERR_ROOM_FULL";

    const assignedSlot = lobby.getFirstAvailableSlot();
    if (assignedSlot === -1) return "ERR_ROOM_FULL";

    const newPlayer: Player = {
      id: assignedSlot,
      tgUserId,
      username,
      isHost: false,
      role: 'crewmate',
      alive: true,
      isGhost: false,
      x: 0, y: 0, lastBroadcastX: 0, lastBroadcastY: 0
    };

    lobby.players.set(assignedSlot, newPlayer);
    this.userToLobbyMap.set(tgUserId, formattedCode);

    console.log(`User ${username} joined lobby ${formattedCode} at slot [${assignedSlot}]`);
    return { lobby, slotId: assignedSlot };
  }

  public handleDisconnect(tgUserId: number): { lobbyCode?: string; destroyed: boolean } {
    const code = this.userToLobbyMap.get(tgUserId);
    if (!code) return { destroyed: false };

    const lobby = this.lobbies.get(code);
    if (!lobby) return { destroyed: false };

    let targetSlot = -1;
    for (const [slotId, player] of lobby.players.entries()) {
      if (player.tgUserId === tgUserId) {
        targetSlot = slotId;
        break;
      }
    }

    if (targetSlot !== -1) {
      lobby.players.delete(targetSlot);
      this.userToLobbyMap.delete(tgUserId);
      console.log(`Evicted slot [${targetSlot}] from lobby ${code}`);
    }

    if (lobby.players.size === 0) {
      this.lobbies.delete(code);
      console.log(`Lobby ${code} empty. Garbage collecting instance.`);
      return { destroyed: true };
    }

    // Execute Host Migration if necessary
    if (lobby.hostSlotId === targetSlot) {
      const remainingSlots = Array.from(lobby.players.keys()).sort((a, b) => a - b);
      const newHostSlot = remainingSlots[0];
      const newHost = lobby.players.get(newHostSlot);
      if (newHost) {
        newHost.isHost = true;
        lobby.hostSlotId = newHostSlot;
        console.log(`Host migrated to slot [${newHostSlot}] in lobby ${code}`);
      }
    }

    return { lobbyCode: code, destroyed: false };
  }
}

// 4. CORE ROUTER & NETWORK INITIALIZATION

const PORT = 8080;
const wss = new WebSocketServer({ port: PORT });
const matchmaker = new LobbyManager();

// Environment-level fallback token configuration
const BOT_TOKEN = process.env.TELEGRAM_BOT_TOKEN || "DEBUG_MOCK_TOKEN";

console.log(`⚡ Authoritative Among Us Binary Server running on port ${PORT}`);

wss.on('connection', (ws: AuthenticatedSocket) => {
  ws.isVerified = false;

  ws.on('message', (rawFrame: Buffer) => {
    // Structural Guard: Enforce strict Node Buffer processing rules
    if (!Buffer.isBuffer(rawFrame) || rawFrame.length === 0) return;

    // Phase 1: Authentication Gateway Validation
    if (!ws.isVerified) {
      const initDataString = rawFrame.toString('utf8');
      const verifiedUser = verifyTelegramAuth(initDataString, BOT_TOKEN);

      if (verifiedUser) {
        ws.isVerified = true;
        ws.tgUserId = verifiedUser.id;
        ws.username = verifiedUser.username || `User_${verifiedUser.id}`;
        console.log(`Cryptographic Handshake Succeeded: ${ws.username}`);
      } else {
        // Validation failure response frame (0x00)
        ws.send(Buffer.from([0x00]));
        return ws.close();
      }
      return;
    }

    // Phase 2: High-Frequency Direct Buffer Packet Unpacking Router
    const opCode = rawFrame.readUint8(0);

    switch (opCode) {
      case 0x10: // Matchmaking Action Handling
        handleMatchmaking(ws, rawFrame);
        break;

      case 0x11: // Direct Spatial Move Synchronization Intent
        handleSpatialMovement(ws, rawFrame);
        break;

      case 0x12: // Authoritative Game Start Execution Command
        handleGameStart(ws);
        break;

      default:
        // Ignore unmapped opcodes silently to protect server loops
        break;
    }
  });

  ws.on('close', () => {
    if (ws.tgUserId) {
      const update = matchmaker.handleDisconnect(ws.tgUserId);
      if (update.lobbyCode && !update.destroyed) {
        broadcastLobbyUpdate(update.lobbyCode);
      }
    }
  });
});

// 5. PACKET ROUTER CORE HANDLER FUNCTIONS
// ==========================================

function handleMatchmaking(ws: AuthenticatedSocket, frame: Buffer) {
  if (frame.length < 2 || !ws.tgUserId || !ws.username) return;
  const subAction = frame.readUint8(1);

  if (subAction === 0x01) { // CREATE LOBBY
    const newLobby = matchmaker.createLobby(ws.tgUserId, ws.username);
    ws.lobbyCode = newLobby.code;
    ws.playerSlotId = 0;

    // Send Handshake Success (0x01) + allocated index code back down pipe
    const response = Buffer.alloc(2);
    response.writeUint8(0x01, 0);
    response.writeUint8(0, 1); // Slot 0
    ws.send(response);

    broadcastLobbyUpdate(newLobby.code);
  } 
  else if (subAction === 0x02) { // JOIN LOBBY
    if (frame.length < 8) return;
    const roomCode = frame.toString('utf8', 2, 8).toUpperCase();
    const joinResult = matchmaker.joinLobby(roomCode, ws.tgUserId, ws.username);

    if (typeof joinResult === "string") {
      // Send Handshake Defeat (0x00) signal
      ws.send(Buffer.from([0x00]));
    } else {
      ws.lobbyCode = joinResult.lobby.code;
      ws.playerSlotId = joinResult.slotId;

      const response = Buffer.alloc(2);
      response.writeUint8(0x01, 0);
      response.writeUint8(joinResult.slotId, 1);
      ws.send(response);

      broadcastLobbyUpdate(joinResult.lobby.code);
    }
  }
}

function handleSpatialMovement(ws: AuthenticatedSocket, frame: Buffer) {
  if (frame.length < 5 || !ws.lobbyCode || ws.playerSlotId === undefined) return;
  const lobby = matchmaker.lobbies.get(ws.lobbyCode);
  if (!lobby || lobby.state !== GameState.ROAMING) return;

  const player = lobby.players.get(ws.playerSlotId);
  if (!player || !player.alive) return;

  // Unpack coordinate calculations safely out of Little-Endian frames
  const rawX = frame.readInt16LE(1);
  const rawY = frame.readInt16LE(3);
  
  // Transform back to coordinate space floats
  const targetedX = rawX / 100;
  const targetedY = rawY / 100;

  // Server Authoritative Verification Rules
  // Future: Check against map collider boundaries here. If safe, write:
  player.x = targetedX;
  player.y = targetedY;
}

function handleGameStart(ws: AuthenticatedSocket) {
  if (!ws.lobbyCode || ws.playerSlotId === undefined) return;
  const lobby = matchmaker.lobbies.get(ws.lobbyCode);
  if (!lobby || lobby.hostSlotId !== ws.playerSlotId || lobby.state !== GameState.LOBBY) return;

  // Lock matchmaking, transition state vectors
  lobby.state = GameState.SPAWN;
  console.log(`Lobby ${lobby.code} launching game state change.`);

  // Implement Role shuffling configuration mapping directly here
  // Inform clients to swap visual layer interfaces...
  const startNotification = Buffer.alloc(2);
  startNotification.writeUint8(0x12, 0); // Opcode Game Start
  startNotification.writeUint8(0x01, 1); // Map ID configuration
  
  broadcastToLobby(lobby.code, startNotification);
}

// ==========================================
// 6. LOW-ALLOCATION BROADCST ROUTINES
// ==========================================

function broadcastLobbyUpdate(lobbyCode: string) {
  const lobby = matchmaker.lobbies.get(lobbyCode);
  if (!lobby) return;

  const usernames = Array.from(lobby.players.values()).map(p => p.username);
  
  // Build 0x10 [0x03] binary message serialization frame
  // Determine dynamically allocated sizing bounds
  let totalPayloadSize = 4; // Opcode + Subcode + Count + HostID
  for (const name of usernames) {
    totalPayloadSize += 1 + Buffer.byteLength(name, 'utf8'); // 1 length prefix byte + content strings
  }

  const broadcastBuffer = Buffer.alloc(totalPayloadSize);
  broadcastBuffer.writeUint8(0x10, 0);
  broadcastBuffer.writeUint8(0x03, 1); // ROOM_UPDATE Sub-Opcode
  broadcastBuffer.writeUint8(lobby.players.size, 2);
  broadcastBuffer.writeUint8(lobby.hostSlotId, 3);

  let writingOffset = 4;
  for (const name of usernames) {
    const byteLen = Buffer.byteLength(name, 'utf8');
    broadcastBuffer.writeUint8(byteLen, writingOffset);
    writingOffset += 1;
    broadcastBuffer.write(name, writingOffset, byteLen, 'utf8');
    writingOffset += byteLen;
  }

  broadcastToLobby(lobbyCode, broadcastBuffer);
}

function broadcastToLobby(lobbyCode: string, payload: Buffer) {
  const lobby = matchmaker.lobbies.get(lobbyCode);
  if (!lobby) return;

  const targetedIds = Array.from(lobby.players.values()).map(p => p.tgUserId);

  wss.clients.forEach((client: AuthenticatedSocket) => {
    if (client.isVerified && client.readyState === WebSocket.OPEN && client.tgUserId && targetedIds.includes(client.tgUserId)) {
      client.send(payload);
    }
  });
}

setInterval(() => {
  for (const lobby of matchmaker.lobbies.values()) {
    if (lobby.state !== GameState.ROAMING) continue;

    const movingSubBlocks: Buffer[] = [];
    
    for (const player of lobby.players.values()) {
      if (!player.alive) continue;

      const dx = player.x - player.lastBroadcastX;
      const dy = player.y - player.lastBroadcastY;
      
      // Strict delta compression check metric expression bounds
      if ((dx * dx + dy * dy) > 0.005) {
        const block = Buffer.alloc(5);
        block.writeUint8(player.id, 0);
        block.writeInt16LE(Math.round(player.x * 100), 1);
        block.writeInt16LE(Math.round(player.y * 100), 3);
        
        movingSubBlocks.push(block);

        // Update local broadcast snapshot coordinates cache
        player.lastBroadcastX = player.x;
        player.lastBroadcastY = player.y;
      }
    }

    // Only dispatch 0xFF packets down structural paths if targets altered positions
    if (movingSubBlocks.length > 0) {
      const deltaPacket = Buffer.alloc(2 + (movingSubBlocks.length * 5));
      deltaPacket.writeUint8(0xFF, 0); // Opcode Delta Sync
      deltaPacket.writeUint8(movingSubBlocks.length, 1);

      let offset = 2;
      for (const block of movingSubBlocks) {
        block.copy(deltaPacket, offset);
        offset += 5;
      }

      broadcastToLobby(lobby.code, deltaPacket);
    }
  }
}, 40); // 25 authoritative ticks per second loop execution rate
